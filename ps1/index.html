def spiral(A):
    n = len(A)
    for i in range(0, n // 2 + 1):
        for j in range(i, n - i):
            print(A[i][j], end=', ')
        for j in range(i + 1, n - i):
            print(A[j][n - i - 1], end=', ')
        for j in range(n - i - 2, i - 1, -1):
            print(A[n - i - 1][j], end=', ')
        for j in range(n - i - 2, i, -1):
            print(A[j][i], end=', ')
    print()

spiral(a)
